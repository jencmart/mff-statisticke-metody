def graph(formula, x_range, label=None):
    """
    Helper function for plotting cook's distance lines
    """
    x = x_range
    y = formula(x)
    plt.plot(x, y, label=label, lw=1, ls='--', color='red')

def diagnostic_plots(df, response, model_fit=None):
    
    X = df.drop([response],  axis=1)
    y = pd.DataFrame(df[response])

    """
    Function to reproduce the 4 base plots of an OLS model in R.

    ---
    Inputs:

    X: A numpy array or pandas dataframe of the features to use in building the linear regression model

    y: A numpy array or pandas series/dataframe of the target variable of the linear regression model

    model_fit [optional]: a statsmodel.api.OLS model after regressing y on X. If not provided, will be
                          generated from X, y
    """

#     if not model_fit:
#         model_fit = sm.OLS(y, sm.add_constant(X)).fit()

    # create dataframe from X, y for easier plot handling
    dataframe = pd.concat([X, y], axis=1)
    # model values
    model_fitted_y = model_fit.fittedvalues
    
    main_figure, axs = plt.subplots(2,2, figsize=(15,10))
    ax1 = axs[0,0]
    ax2 = axs[0,1]
    ax3 = axs[1,0]
    ax4 = axs[1,1]

    
    # ############### 1. Residuals X Fitted #####################################################
    # model residuals
    model_residuals = model_fit.resid
    # absolute residuals
    model_abs_resid = np.abs(model_residuals)
    # Plot
    sns.residplot(model_fitted_y, dataframe.columns[-1], data=dataframe, lowess=True, scatter_kws={'alpha': 0.5}, line_kws={'color': 'red', 'lw': 1, 'alpha': 0.8}, ax=ax1)
    # Labels
    ax1.set_title('Residuals vs Fitted')
    ax1.set_xlabel('Fitted values')
    ax1.set_ylabel('Residuals')
    # Annotations
    abs_resid = model_abs_resid.sort_values(ascending=False)
    abs_resid_top_3 = abs_resid[:3]
    for i in abs_resid_top_3.index:
        ax1.annotate(i, xy=(model_fitted_y[i], model_residuals[i]));

    
    
    # ############### 2. QQ #####################################################   
    # normalized residuals
    model_norm_residuals = model_fit.get_influence().resid_studentized_internal
    # QQ plot
    QQ = ProbPlot(model_norm_residuals)
    plot_lm_2 = QQ.qqplot(line='45', alpha=0.5, color='#4C72B0', lw=1, ax=ax2)
    # Labels
    ax2.set_title('Normal Q-Q')
    ax2.set_xlabel('Theoretical Quantiles')
    ax2.set_ylabel('Standardized Residuals');
    # Annotations
    abs_norm_resid = np.flip(np.argsort(np.abs(model_norm_residuals)), 0)
    abs_norm_resid_top_3 = abs_norm_resid[:3]
    for r, i in enumerate(abs_norm_resid_top_3):
        ax2.annotate(i, xy=(np.flip(QQ.theoretical_quantiles, 0)[r], model_norm_residuals[i]));

    # ############### 3. Scale X Location #####################################################

    # absolute squared normalized residuals
    model_norm_residuals_abs_sqrt = np.sqrt(np.abs(model_norm_residuals))
    # Plot
    ax3.scatter(model_fitted_y, model_norm_residuals_abs_sqrt, alpha=0.5);
    sns.regplot(model_fitted_y, model_norm_residuals_abs_sqrt, scatter=False, ci=False, lowess=True, line_kws={'color': 'red', 'lw': 1, 'alpha': 0.8}, ax=ax3);
    # Labels
    ax3.set_title('Scale-Location')
    ax3.set_xlabel('Fitted values')
    ax3.set_ylabel('$\sqrt{|Standardized Residuals|}$');
    # Annotations
    abs_sq_norm_resid = np.flip(np.argsort(model_norm_residuals_abs_sqrt), 0)
    abs_sq_norm_resid_top_3 = abs_sq_norm_resid[:3]
    for i in abs_norm_resid_top_3:
        ax3.annotate(i, xy=(model_fitted_y[i], model_norm_residuals_abs_sqrt[i]));


    # ############### 4. Residuals X Leverage #####################################################   
    # leverage, from statsmodels internals
    model_leverage = model_fit.get_influence().hat_matrix_diag
    # Plot
    ax4.scatter(model_leverage, model_norm_residuals, alpha=0.5);
    sns.regplot(model_leverage, model_norm_residuals, scatter=False, ci=False, lowess=True, line_kws={'color': 'red', 'lw': 1, 'alpha': 0.8}, ax=ax4);
    # Labels
    ax4.set_xlim(0, max(model_leverage)+0.01)
    ax4.set_ylim(-3, 5)
    ax4.set_title('Residuals vs Leverage')
    ax4.set_xlabel('Leverage')
    ax4.set_ylabel('Standardized Residuals');
    # cook's distance, from statsmodels internals
    model_cooks = model_fit.get_influence().cooks_distance[0]
    # Annotations
    leverage_top_3 = np.flip(np.argsort(model_cooks), 0)[:3]
    for i in leverage_top_3:
        ax4.annotate(i, xy=(model_leverage[i], model_norm_residuals[i]));
    p = len(model_fit.params) # number of model parameters
    graph(lambda x: np.sqrt((0.5 * p * (1 - x)) / x), np.linspace(0.001, max(model_leverage), 50), 'Cook\'s distance') # 0.5 line
    graph(lambda x: np.sqrt((1 * p * (1 - x)) / x), np.linspace(0.001, max(model_leverage), 50)) # 1 line
    ax4.legend(loc='upper right');

def brute_force_select(data, response, adj_r_threshold):
    # create combinations 
    lst = data.columns.tolist()
    lst.remove(response)
    combs = []
    for i in range(1, len(lst)+1):
        els = [list(x) for x in itertools.combinations(lst, i)]
        combs.extend(els)
        
    # filter combs
    filtered_combs = []
    for comb in combs:
        bad = False
        for x in lst:
            if (x + "_pow2" in comb and not x in comb) or  (x in comb and  x + "_log" in comb) or (x in comb and  x + "_sqrt" in comb) or (x + "_log" in comb and  x + "_pow2" in comb) or (x + "_log" in comb and  x + "_sqrt" in comb) or (x + "_pow2" in comb and x + "_sqrt" in comb):
                bad = True
        if not bad:
            filtered_combs.append(comb)
    
    # fit all combinations
    results = []
    print("fitting {} variants".format(len(filtered_combs)))
    for i, comb in enumerate(filtered_combs):             
        formula = "{} ~ {}".format(response, ' + '.join(comb))
#         if i > 0 and i%1000 == 0:
#             print("{}/{}: {}".format(i+1, len(filtered_combs), formula))
        model = smf.ols(formula, data).fit()
        score = model.rsquared_adj
        if score > adj_r_threshold:
            results.append((score, model))

    # sort
    sorted_res = sorted(results,  key=lambda res: res[0], reverse=True)
    return sorted_res
